# LLM Agent 代码审查系统 - 依赖清单
# 安装：pip install -r requirements.txt

# ==================== 核心依赖 ====================
python-dotenv        # 环境变量加载
httpx                # 异步 HTTP 客户端（LLM API 调用）
openai               # OpenAI 兼容 API 客户端
pydantic             # 数据验证与序列化

# ==================== Diff 解析 ====================
unidiff              # Git diff 解析

# ==================== Web 服务 ====================
fastapi              # API 框架
uvicorn              # ASGI 服务器
starlette            # FastAPI 底层依赖

# ==================== 配置与工具 ====================
pyyaml               # YAML 配置解析
psutil               # 系统进程管理

# ==================== 代码扫描器 ====================
semgrep              # 多语言通用静态分析（Python/JS/TS/Java/Go/Ruby/C/C++等）
# 以下为可选的语言特定扫描器（需安装对应语言环境）
# pylint             # Python 静态分析
# flake8             # Python 代码风格检查
# mypy               # Python 类型检查

# ==================== GUI（可选）====================
ttkbootstrap         # 现代化 Tkinter GUI

# ==================== 测试 ====================
pytest               # 单元测试框架
hypothesis           # 属性测试框架
